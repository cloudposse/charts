#
# Terraform Resources to Support Kube Backups to S3
#

#
# Provide S3 bucket to store backups
# 
resource "aws_s3_bucket" "{{ template "s3bucket" . }}" {
  bucket = "{{ template "s3bucket" . }}"
  acl    = "private"

  tags {
    Name      = "{{ template "s3bucket" . }}"
    Release   = "{{ .Release.Name }}"
    Component = "backup"
    App       = "{{ template "fullname" . }}"
    Chart     = "{{ .Chart.Name }}-{{ .Chart.Version | replace "+" "_" }}"
  }
}

# 
# Provide IAM Profile with sufficient permissions to upload backups to S3 bucket
# 
variable "base_role_arn" {
  type = "string"
}

resource "aws_iam_role" "{{ template "iam_role" . }}" {
  name = "{{ template "iam_role" . }}"

  assume_role_policy = <<EOF
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Action": "sts:AssumeRole",
      "Principal": {
        "Service": "ec2.amazonaws.com"
      },
      "Effect": "Allow"
    },
    {
      "Action": "sts:AssumeRole",
      "Principal": {
        "AWS": "${var.base_role_arn}"
      },
      "Effect": "Allow"
    }
  ]
}
EOF
}

resource "aws_iam_role_policy" "AccessToS3-{{ template "iam_role" . }}" {
  name = "AccessToS3-{{ template "iam_role" . }}"
  role = "${aws_iam_role.{{ template "iam_role" . }}}"

  policy = <<EOF
{
  "Version": "2012-10-17",
  "Statement": [
          {
              "Effect": "Allow",
              "Action": [
                  "s3:ListBucket"
              ],
              "Resource": [
                  "arn:aws:s3:::{{ template "s3bucket" . }}"
              ]
          },
          {
              "Effect": "Allow",
              "Action": [
                  "s3:PutObject",
                  "s3:GetObject",
                  "s3:DeleteObject"
              ],
              "Resource": [
                  "arn:aws:s3:::{{ template "s3bucket" . }}/*"
              ]
          }
      ]
}
EOF
}
